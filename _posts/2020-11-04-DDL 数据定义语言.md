## DDL数据定义语言

数据定义语言，库和表的管理

库的管理：创建、修改、删除

表管理：创建、修改、删除

创建：create，修改 alter，删除 drop

## 一、库管理

```sql
/**
一、库管理
1、创建库
create database 库名：
 */
create database  if not exists books;# 一般是如此配合使用

/**
2、修改库
 */
# rename database 库名 to 新库名
# alter database girls character set utf-8;
# drop  database if exists 库名
drop database if exists books;
```

## 二、表管理

```sql
/**
二、表管理
1、创建表名

create table 表名（
    列名 列的类型 【长度 约束】，
    列名 列的类型 【长度 约束】，
    列名 列的类型 【长度 约束】，
    ......
）

 */

create table if not exists book (
  id          int,
  bName       varchar(20),
  price       double,
  authorId    int,
  publishDate datetime

);

desc book;

create table if not exists author (
  id      int,
  au_name varchar(20),
  nation  varchar(20)
);

desc author;
/**
2、表修改
语法：
alter table 表名 add/drop/modify/changer column 列名【类型、约束】
 */
# 修改列名
# 1、修改列名
alter table book
  change column publishDate pubDate DATETIME;

# 2、修改累的类型或约束
alter table book
  modify column pubDate timestamp;

# 3、添加新的列
alter table author
  add column annual double;

# 4、删除列
alter table author
  drop column annual;

# 5、修改表名
alter table auth
rename to book_author;

desc book_author;

# 6、 删除表
drop table if exists book_author;
show tables;

/**
3、表复制
 */
insert into book_author values (1, "村上", "日本"),
  (2, "莫言", "中国"), (1, "卡斯特罗", "古巴");

/**
1、仅仅复制表的结构
 */
create table copy_book_author
  like book_author;

# 仅仅复制表结构的部分字段，的结构
create table copy_book_author1
    select
      id,
      au_name
    from book_author
    where 1 = 2;# 设置一个谁都无法满足的条件

/**
2、复制表结构+数据
 */
create table copy_book_author2
    select *
    from book_author;

/**
3 、复制部分数据
 */
create table copy_book_author3
    select
      id,
      au_name,
      nation
    from book_author
    where book_author.nation = '日本';

select *
from copy_book_author3;
```
